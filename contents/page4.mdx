```javascript
 it('メッセージに「餃子に」が含まれていて、数値がある場合、数値の内容をfirebaseに保存して、メッセージ「🥟から1,000円を入れるね」を返すこと', async () => {
    const app = await setup()
    const message = '餃子に1000'
    const res: any = await request(app)
      .post('/bot/webhook')
      .send({ message })
    expect(res.body).toEqual([
      {
        message: {
          text: '🥟に1,000円を入れるね',
          type: 'text',
        },
      },
    ])
  })
```
